// Original example - http://learningprocessing.com/examples/chp09/example-09-11-array-append //Resizing an array using append() var balls = []; // We start with an array with just one element. var gravity = 0.1; function setup() { createCanvas(480, 240); // Initialize ball index 0 balls.push(new Ball(50,0,32,random(0,255),random(0,255),random(0,255))); } function draw() { background(51); // Update and display all balls for (var i = 0; i < balls.length; i ++ ) { // Whatever the length of that array, update and display all of the objects. balls[i].update(); balls[i].display(); } } function mousePressed() { } function keyPressed() { fill(random(0,255),random(0,255),random(0,255)) let r = random(0,255); let g = random(0,255); let b_ = random(0,255); if (keyCode === 37) { //console.log("left"); var b = new Ball(mouseX,mouseY,32,r,g,b_); // Make a new object at the mouse location. balls.push(b); } else if (keyCode === 38) { //console.log("up"); balls.pop(); } else if (keyCode === 39) { //console.log("right"); var b = new Ball(mouseX,mouseY,32,r,g,b_); // Make a new object at the mouse location. balls.unshift(b); } else if (keyCode === 40) { //console.log("down"); balls.shift(); } else {} } // Ball constuctor function Ball(tempX, tempY, tempW, tempR, tempG, tempB) { this.x = tempX; // x location of square this.y = tempY; // y location of square this.w = tempW; // speed of square this.speed = 0; // size this.r = tempR; this.g = tempG; this.b = tempB; this.display = function() { // Display the square fill(this.r,this.g,this.b); stroke(0); ellipse(this.x,this.y,this.w,this.w); } this.update = function() { // Add speed to location this.y = this.y + this.speed; // Add gravity to speed this.speed = this.speed + gravity; // If square reaches the bottom // Reverse speed if (this.y > height) { this.y = height; this.speed = this.speed * -0.95; } } }
